1.	Write a program to check whether a given string is a palindrome or not using for loop and if else statement.

import java.util.Scanner;
public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner (System.in);
        int number = sc.nextInt();
        int temp= number;
        int add=0;
        while(number>0)
        {
            int rem = number % 10;
            add = (add * 10)+rem;
            number = number /10;
            
        }
        if (add == temp)
        {
            System.out.println("Palindrome:"+temp);
        }
        else
        {
            System.out.println("Not a palindrome:"+temp);
        }
    }
}
2.Write a program to that reads a string from user and display the reversed string using loop.
import java.util.Scanner;
public class Main {
    public static void main(String[] args) {
       Scanner sc = new Scanner (System.in);
       String name = sc.nextLine();
       String reverse=" ";
       char temp;
       for (int i=0;i<name.length();i++)
       {
        temp= name.charAt(i);
        reverse= temp+reverse;
       }
       
       System.out.println("Reversed string is :"+reverse);
    }
}

public class Main {
    public static void main(String[] args) {
        Scanner sc= new Scanner (System.in);
        int n = sc.nextInt();
        int temp=1;
        for (int i=1;i<=n;i++)
        {
          for (int j=1;j<=i;j++)
          {
              System.out.print(temp+++"\t");
          }
          
          System.out.println();
        }
    }

    public static void main(String[] args) {
        Scanner sc= new Scanner (System.in);
        int n = sc.nextInt();
        int k=n*2-1;
        for (int i=1;i<=k;i++)
        {
          for (int j=1;j<=k;j++)
          {
              if (j==i || j==k-i+1)
              System.out.print("*"+"\t");
          }
          
          System.out.println();

    }
}
